241  if (reply == QMessageBox::Yes) {
242  foreach(QGraphicsItem *item, scene()->selectedItems()) {
243  if(items().indexOf(item) != -1) {
244  Bar *bar = dynamic_cast<Bar *> (item);
245 
246  if(bar != NULL) {
247  foreach(Line *line, bar->lines) {
248  scene()->removeItem(line);
249  line->network()->removeLine(line);
250  delete line;
251  }
