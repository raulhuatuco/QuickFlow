185 /*******************************************************************************

188 void SystemView::mouseMoveEvent(QMouseEvent *event)
189 {
190  if (event->buttons().testFlag(Qt::MidButton)) {
191  QScrollBar *hBar = horizontalScrollBar();
192  QScrollBar *vBar = verticalScrollBar();
193  QPoint delta = event->pos() - oldPos;
194  oldPos = event->pos();
195  hBar->setValue(hBar->value() + (isRightToLeft() ? delta.x() : -delta.x()));
196  vBar->setValue(vBar->value() - delta.y());
197  }
198 
199  QGraphicsView::mouseMoveEvent(event);
200 }
