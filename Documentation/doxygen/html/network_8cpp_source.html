120 /*******************************************************************************
121  * Add line.
122  ******************************************************************************/
123 bool Network::addLine(Line *line)
124 {
125  // Check if line exists.
126  if(getLineByNodes(line->nodes) != NULL)
127  return false;
128 
129  // Check if nodes exists.
130  Bar *pNoI = getBarById(line->nodes.first);
131  Bar *pNoF = getBarById(line->nodes.second);
132 
133  if ((pNoI == NULL) || (pNoF == NULL))
134  return false;
135 
136  line->setNetwork(this);
137 
138  line->setNodes(pNoI, pNoF);
139 
140  lines.insert(line->nodes, line);
141 
142  return true;
143 }

196 QJsonObject Network::toJson()
197 {
198  QJsonObject netJson;
199 
200  netJson.insert("name", name);
201  netJson.insert("xOffset", xOffset);
202  netJson.insert("yOffset", yOffset);
203  netJson.insert("barStrokeColor", barContourColor.name());
204  netJson.insert("barSlackFillColor", slackBarColor.name());
205  netJson.insert("barPqFillColor", pqBarColor.name());
206  netJson.insert("lineColor", lineColor.name());
207  netJson.insert("voltageBase", voltageBase_);
208  netJson.insert("powerBase", powerBase_);
209 
210  QJsonArray barArray;
211  foreach(Bar *bar,bars) {
212  barArray << bar->toJson();
213  }
214  netJson.insert("barArray", barArray);
215 
216  QJsonArray lineArray;
217  foreach(Line *line, lines) {
218  lineArray << line->toJson();
219  }
220  netJson.insert("lineArray", lineArray);
221 
222  return netJson;
223 }

225 /*******************************************************************************

228 void Network::fromJson(QJsonObject &netJson)
229 {
230  name = netJson.value("name").toString();
231  xOffset = netJson.value("xOffset").toDouble();
232  yOffset = netJson.value("yOffset").toDouble();
233  barContourColor.setNamedColor(netJson.value("barStrokeColor").toString());
234  slackBarColor.setNamedColor(netJson.value("barSlackFillColor").toString());
235  pqBarColor.setNamedColor(netJson.value("barPqFillColor").toString());
236  lineColor.setNamedColor(netJson.value("lineColor").toString());
237  voltageBase_ = netJson.value("voltageBase").toDouble();
238  powerBase_ = netJson.value("powerBase").toDouble();
239 
240  QJsonArray barArray(netJson.value("barArray").toArray());
241 
242  foreach(QJsonValue barValue, barArray) {
243  QJsonObject jsonBar(barValue.toObject());
244 
245  Bar *bar = new Bar;
246  bar->fromJson(jsonBar);
247  addBar(bar);
248  }
249 
250  QJsonArray lineArray;

252
