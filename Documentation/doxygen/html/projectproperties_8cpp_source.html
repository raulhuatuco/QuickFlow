190 /*******************************************************************************
191  * Button Box accepted.
192  ******************************************************************************/
193 void ProjectProperties::on_buttonBox_accepted()
194 {
195  // Check for empty name.
196  if (ui->name->text().isEmpty()) {
197  QMessageBox::information(this, "Invalid name", "Invalid project name",
198  QMessageBox::Ok);
199  ui->name->setFocus();
200  return;
201  }
202 
203  // Check for empty path.
204  if (ui->path->text().isEmpty()) {
205  QMessageBox::information(this, "Invalid path", "Invalid project path",
206  QMessageBox::Ok);
207  ui->path->setFocus();
208  return;
209  }
210 
211  // Check for empty max iterations.
212  if(ui->maxIterations->text().isEmpty()) {
213  QMessageBox::information(this, "Invalid max iterations",
214  "Select a maximum number of iterations.",
215  QMessageBox::Ok);
216  ui->maxIterations->setFocus();
217  return;
218  }
219 
220  // Check for empty min error.
221  if(ui->minError->text().isEmpty()) {
222  QMessageBox::information(this, "Invalid min error",
223  "Select a minimum error.",
224  QMessageBox::Ok);
225  ui->minError->setFocus();
226  return;
227  }
228 
229  // Fill project data.
230  project_->name = ui->name->text();
231  project_->filePath = ui->path->text() + "/" + project_->name + ".qkf";
232  project_->setMaxIterations(ui->maxIterations->text().toUInt());
233  project_->setMinError(ui->minError->text().toDouble()/100.0);
234  project_->setLengthUn(ui->lengthUnit->currentData().value<Unit::LengthUnit>());
235  project_->setImpedanceUn(
236  ui->impedanceUnit->currentData().value<Unit::ImpedanceUnit>());
237  project_->setVoltageUn(
238  ui->voltageUnit->currentData().value<Unit::VoltageUnit>());
239  project_->setPowerUn(ui->powerUnit->currentData().value<Unit::PowerUnit>());
240  project_->setCurrentUnit(
241  ui->currentUnit->currentData().value<Unit::CurrentUnit>());
242 
243  // Try to save project.
244  bool save_ok = project_->save();
245 
246  if (!save_ok) {
247  QMessageBox::critical(this, "File write error",
248  "File write error, please choose another location!",
249  QMessageBox::Ok);
250  return;
251  }
252 
253  accept();
254 }
