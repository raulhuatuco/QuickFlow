139 /*******************************************************************************
140 * Create OGDF graph.
141 *******************************************************************************/
142 void RedrawNetwork::createGraph()
143 {
144  graph = new Graph;
145  graphAttributes = new GraphAttributes(*graph, GraphAttributes::nodeGraphics |
146  GraphAttributes::edgeGraphics |
147  GraphAttributes::edgeType |
148  GraphAttributes::edgeArrow);
149  graphAttributes->setDirected(true);
150 
151  node n;
152 
153  foreach(Bar *bar, network_->bars) {
154  n = graph->newNode(bar->id());
155  nodeMap.insert(bar->id(),n);
156  }
157 
158  forall_nodes(n, *graph) {
159  graphAttributes->height(n) = Network::barIconSize;
160  graphAttributes->width(n) = Network::barIconSize;
161  graphAttributes->shape(n) = shRect;
162  }
163 
164  foreach(Line *line, network_->lines) {
165  node noI = nodeMap.value(line->nodes.first);
166  node noF = nodeMap.value(line->nodes.second);
167  graph->newEdge(noI, noF);
168  }
169 
170  edge e;
171  forall_edges(e, *graph) {
172  graphAttributes->arrowType(e) = eaNone;
173  }
174 }

179 void RedrawNetwork::applyChanges()

182

184  double x = graphAttributes->x(n);
185  double y = graphAttributes->y(n);
186  int id = n->index();
187  Bar *bar = network_->getBarById(id);
188 
189  bar->setPos(y, x);

193 /*******************************************************************************

196 void RedrawNetwork::removeOffset()
197 {
198  Bar *slack = network_->getBarById(0);
199  double xOffset = slack->x();
200  double yOffset = slack->y();
201  slack->setX(0);
202  slack->setY(0);
203 
204  foreach(Bar *bar, network_->bars) {
205  if(bar->id() != 0) {
206  bar->setX(bar->x() - xOffset);
207  bar->setY(bar->y() - yOffset);
208  }
209  }
210 }
