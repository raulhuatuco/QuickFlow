68 /*******************************************************************************
69  * Solve network.
70  ******************************************************************************/
71 bool Shirmoharmmadi::solve()
72 {
73  // Create a timer to measure the duration and start it.
74  QElapsedTimer timer;
75  timer.start();
76 
77  // Set Initial voltage to 1pu.
78  double initialVoltage = network_->voltageBase();
79  foreach(Bar *bar, network_->bars) {
80  bar->setV(0, std::polar(initialVoltage, 0.0), Unit::kVolts);
81  bar->setV(1, std::polar(initialVoltage, 120.0*kPI/180.0), Unit::kVolts);
82  bar->setV(2, std::polar(initialVoltage, 240.0*kPI/180.0), Unit::kVolts);
83  }
84 
85  // Initialize at iteration 0 and error inf;
86  uint32_t iteration = 0;
87  double iterationError = qInf();
88 
89  // Create a radial layer to operate the algorithm.
90  RadialLayer radLayer(network_);
91  radLayer.calculate();
92 
93  do {
94  // Calculates the backward sweep.
95  doBackwardSweep(radLayer);
96 
97  // Calculates the forward sweep.
98  doForwardSweep(radLayer);
99 
100  // Calculate Slack current.
101  computeSlackCurrent();
102 
103  // Calculate the iteration error.
104  iterationError = maxError();
105 
106  iteration++;
107  } while(( iterationError > network_->minError) &&
108  (iteration < network_->maxIterations));
109 
110  duration = timer.elapsed();
111  usedIterations = iteration;
112 
113  if (iteration < network_->maxIterations)
114  return true;
115  else
116  return false;
117 }

119 /*******************************************************************************
120  * Backward sweep.
121  ******************************************************************************/
122 void Shirmoharmmadi::doBackwardSweep(RadialLayer &radLayer)
123 {
124  // Creates a list to mark lines that has beed processed.
125  QList<Line *> processedLines;
126 
127  //Backward sweep start here.
128  // Calculate line current.
129  for(int i = radLayer.width() -1; i>0; i--) {
130 
131  // Process every bar in the layer.
132  foreach(Bar *bar, *radLayer.layers.at(i)) {
133 
134  // Process every line of the bar.
135  foreach(Line *line, bar->lines) {
136 
137  // Check if line already have I.
138  if(processedLines.indexOf(line) == -1) {
139 
140  // If not, sum every other lines I.
141  complex<double> iLine[3];
142 
143  foreach(Line *oline, bar->lines) {
144  if(oline != line) {
145  iLine[0] += oline->i(0, Unit::kAmperePerUnit);
146  iLine[1] += oline->i(1, Unit::kAmperePerUnit);
147  iLine[2] += oline->i(2, Unit::kAmperePerUnit);
148  }
149  }
150 
151  // Remove Bar I.
152  iLine[0] -= bar->rI(0, Unit::kAmperePerUnit);
153  iLine[1] -= bar->rI(1, Unit::kAmperePerUnit);
154  iLine[2] -= bar->rI(2, Unit::kAmperePerUnit);
155 
156  // Set line current.
157  line->setI(0, iLine[0], Unit::kAmperePerUnit);
158  line->setI(1, iLine[1], Unit::kAmperePerUnit);
159  line->setI(2, iLine[2], Unit::kAmperePerUnit);
160  processedLines.append(line);
161  }
162  }
163  }
164  }
165 }

167 /*******************************************************************************
168  * Forward sweep.
169  ******************************************************************************/
170 void Shirmoharmmadi::doForwardSweep(RadialLayer &radLayer)
171 {
172 // Creates a list to mark lines that has beed processed.
173  QList<Line *> processedLines;
174 
175  // Forward sweep.
176  for(int i=0; i<radLayer.width() -1; i++) {
177 
178  // Go trough every bar.
179  foreach(Bar *bar, *radLayer.layers[i]) {
180 
181  // Process every line of the bar.
182  foreach(Line *line, bar->lines) {
183 
184  // Check if line has been processed.
185  if(processedLines.indexOf(line) == -1) {
186 
187  complex<double> bVoltage[3];
188  bVoltage[0] = bar->rV(0,Unit::kVoltsPerUnit);
189  bVoltage[1] = bar->rV(1,Unit::kVoltsPerUnit);
190  bVoltage[2] = bar->rV(2,Unit::kVoltsPerUnit);
191 
192  bVoltage[0] -= line->z(0, Unit::kOhmPerUnit)*line->i(0,Unit::kAmperePerUnit);
193  bVoltage[0] -= line->z(1, Unit::kOhmPerUnit)*line->i(1, Unit::kAmperePerUnit);
194  bVoltage[0] -= line->z(2,Unit::kOhmPerUnit)*line->i(2, Unit::kAmperePerUnit);
195 
196  bVoltage[1] -= line->z(1, Unit::kOhmPerUnit)*line->i(0,Unit::kAmperePerUnit);
197  bVoltage[1] -= line->z(3, Unit::kOhmPerUnit)*line->i(1, Unit::kAmperePerUnit);
198  bVoltage[1] -= line->z(4,Unit::kOhmPerUnit)*line->i(2, Unit::kAmperePerUnit);
199 
200  bVoltage[2] -= line->z(2, Unit::kOhmPerUnit)*line->i(0,Unit::kAmperePerUnit);
201  bVoltage[2] -= line->z(4, Unit::kOhmPerUnit)*line->i(1, Unit::kAmperePerUnit);
202  bVoltage[2] -= line->z(5,Unit::kOhmPerUnit)*line->i(2, Unit::kAmperePerUnit);
203 
204  line->pNoF()->setRV(0, bVoltage[0], Unit::kVoltsPerUnit);
205  line->pNoF()->setRV(1, bVoltage[1], Unit::kVoltsPerUnit);
206  line->pNoF()->setRV(2, bVoltage[2], Unit::kVoltsPerUnit);
207 
208  processedLines.append(line);
209  }
210  }
211  }
212  }

215 /*******************************************************************************
216  * Compute slack current.
217  ******************************************************************************/
218 void Shirmoharmmadi::computeSlackCurrent()
219 {
220  Bar *slack = network_->getBarById(0);
221 
222  complex<double> iSlack[3];
223 
224  foreach(Line *sline, slack->lines) {
225  iSlack[0] -= sline->i(0, Unit::kAmperePerUnit);
226  iSlack[1] -= sline->i(1, Unit::kAmperePerUnit);
227  iSlack[2] -= sline->i(2, Unit::kAmperePerUnit);
228  }
229 
230  complex<double> siSlack[3];
231  siSlack[0] = slack->rV(0)*conj(iSlack[0]);
232  siSlack[1] = slack->rV(1)*conj(iSlack[1]);
233  siSlack[2] = slack->rV(2)*conj(iSlack[2]);
234 
235  slack->setSi(0, siSlack[0]);
236  slack->setSi(1, siSlack[2]);
237  slack->setSi(2, siSlack[1]);
238 }

242  ******************************************************************************/
243 double Shirmoharmmadi::maxError()
244 {
245  // Holds the last maximum error.
246  double maxError = 0.0;
247 
248  Bar *slack = network_->getBarById(0);
249 
250  complex<double> deltaS[3];
251 
252  deltaS[0] = slack->si(0, Unit::kVA) - oldSlackPower[0];
253  deltaS[1] = slack->si(1, Unit::kVA) - oldSlackPower[1];
254  deltaS[2] = slack->si(2, Unit::kVA) - oldSlackPower[2];
255 
256  for (int i=0; i<3; i++) {
257  double realError;
258  double imagError;
259 
260  realError = abs(deltaS[i].real());
261  imagError = abs(deltaS[i].imag());
262 
263  if(realError > maxError) {
264  maxError = realError;
265  }
266 
267  if(imagError > maxError) {
268  maxError = imagError;
269  }
270  }
271 
272  oldSlackPower[0] = slack->si(0, Unit::kVA);
273  oldSlackPower[1] = slack->si(1, Unit::kVA);
274  oldSlackPower[2] = slack->si(2, Unit::kVA);
275 
276  return maxError;
277 }
