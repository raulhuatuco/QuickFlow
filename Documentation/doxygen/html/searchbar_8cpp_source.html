95 void SearchBar::on_search_clicked()
96 {
97  // Check for empty data.
98  if(ui->id->text().isEmpty()) {
99  QMessageBox::information(this, "Invalid parameter", "Parameter ID is empty.",
100  QMessageBox::Ok);
101  ui->id->setFocus();
102  return;
103  }
104 
105  // Check for empty data.
106  if(ui->networks->currentIndex() == -1) {
107  QMessageBox::information(this, "Invalid parameter",
108  "Parameter network is invalid.",
109  QMessageBox::Ok);
110  ui->networks->setFocus();
111  return;
112  }
113 
114  // Search bar.
115  Network *network = project_->networks.value(ui->networks->currentText());
116  bar_ = network->getBarById(ui->id->text().toInt());
117 
118  if(bar_ == NULL) {
119  QMessageBox::information(this, "Invalid parameter",
120  "Bar not found.",
121  QMessageBox::Ok);
122  ui->id->setFocus();
123  return;
124  }
125 
126  accept();
127 }
