84 void NetworkProperties::setOptions(Project *project, Network *network)
85 {
86 // Adjust apearance according to network.
87  if (network == NULL) {
88  setWindowTitle(tr("New network"));
89  network_ = new Network;
90  isNew = true;
91 
92  } else {
93  setWindowTitle(tr("Edit network ") + network->name);
94  network_ = network;
95  isNew = false;
96 
97  // Initialize colors from network.
98  QString slackColor =
99  QString("border: 2px solid #000000;" \
100  "\nborder-radius: 6px;" \
101  "\nbackground-color: %1;" \
102  "\nselection-background-color: %2;").arg(

111  network->pqBarColor.name()).arg(network->pqBarColor.name());
112  ui->pqColor->setStyleSheet(pqColor);
113 
114  QString lineColor =
115  QString("border: 2px solid #000000;" \
116  "\nborder-radius: 6px;" \
117  "\nbackground-color: %1;" \
118  "\nselection-background-color: %2;").arg(
119  network->lineColor.name()).arg(network->lineColor.name());
120  ui->lineColor->setStyleSheet(lineColor);
121 
122  QString borderColor =
123  QString("border: 2px solid #000000;" \
124  "\nborder-radius: 6px;" \
125  "\nbackground-color: %1;" \
126  "\nselection-background-color: %2;").arg(
127  network->barContourColor.name()).arg(network->barContourColor.name());
128  ui->barBorderColor->setStyleSheet(borderColor);
129 
130  // Initialize bases.
131  // Note that the input value is the line voltage, so we neeed to multiply
132  // the stored voltage base by sqrt(3), because the software uses the phase
133  // voltage in the calculations:
134  // Vline = Vline*sqrt(3);
135  ui->voltageBase->setText(QString::number(network->voltageBase()*kSQRT3));
136 
137  // Note that the power base input is the total power. We use the power per
138  // fase, so, to display the correct value we need to get the total power,
139  // wich is:
140  // Ptotal = Pphase*3.0;
141  ui->powerBase->setText(QString::number(network->powerBase()*3.0));
142  }
143 
144  // Store project parameters.
145  project_ = project;
146 }

240 /*******************************************************************************
241  * Button accept.
242  ******************************************************************************/
243 void NetworkProperties::on_buttonBox_accepted()
244 {
245  // Check for empty fields.
246  if(ui->networkName->text().isEmpty()) {
247  QMessageBox::information(this, "Invalid parameter", "Network name is empty.",
248  QMessageBox::Ok);
249  ui->networkName->setFocus();
250  return;
251  }
252 
253  if(ui->xOffset->text().isEmpty()) {
254  ui->xOffset->setText("0");
255  }
256 
257  if(ui->yOffset->text().isEmpty()) {
258  ui->yOffset->setText("0");
259  }
260 
261  if(ui->voltageBase->text().isEmpty()) {
262  QMessageBox::information(this, "Invalid voltage base",
263  "Select a voltage base.",
264  QMessageBox::Ok);
265  ui->voltageBase->setFocus();
266  return;
267  }
268 
269  if(ui->powerBase->text().isEmpty()) {
270  QMessageBox::information(this, "Invalid power base",
271  "Select a power base.",
272  QMessageBox::Ok);
273  ui->powerBase->setFocus();
274  return;
275  }
276 
277  // Fill network data.
278  network_->name = ui->networkName->text();
279  network_->slackBarColor = slack;
280  network_->pqBarColor = pq;
281  network_->barContourColor = barBorder;
282  network_->lineColor = line;
283  network_->xOffset = ui->xOffset->text().toDouble();
284  network_->yOffset = ui->yOffset->text().toDouble();
285 
286  // Note that the input value is the line voltage, so we neeed to divide it
287  // by sqrt(3), because the software uses the phase voltage in the
288  // calculations:
289  // Vphase = Vline/sqrt(3);
290  network_->setVoltageBase(ui->voltageBase->text().toDouble()/kSQRT3);
291 
292 
293  // Note that the power base input is the total power. We use the power per
294  // fase, so, since we have 3 phases, Pphase = Ptotal/3.0;
295  network_->setPowerBase(ui->powerBase->text().toDouble()/3.0);
296 
297  // If new, we need to insert the network into the project.
298  if(isNew)
299  project_->networks.insert(network_->name, network_);
300 
301  accept();
302 }
